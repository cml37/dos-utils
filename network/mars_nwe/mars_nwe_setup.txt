Build Mars NWE from Source
==================================================
sudo apt-get install libgdbm-dev
wget ftp://ftp.koansoftware.com/public/opensource/mars_nwe/mars_nwe-0.99.pl21.tgz
tar xzf mars_nwe-0.99.pl21.tgz
cd mars_nwe
wget https://github.com/cml37/dos-utils/blob/master/network/mars_nwe/patch/mars_fix.diff
patch -i mars_fix.diff
make
make install
